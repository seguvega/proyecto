create or replace function carga_th_defunciones_fetales (vanio int,vtrimestre int)
return number
is 
    cursor c_th_tiempo is 
    select * from d_tiempo 
        where anio = vanio
        and trimestre= vtrimestre
    order by id_dimt;

    cursor c_th_ubic is 
    select ID_DIMUBIC, id_provincia, id_canton, id_parroquia 
    from d_ubic_geo
    where id_provincia <> 90
    order by id_dimubic;

    cursor c_th_etnia is 
    select id_dimetnia, id_etnia,nombre_etnia 
    from d_etnia 
    order by id_dimetnia ;


    --------------- definición de registros
    r_tiempo c_th_tiempo%rowtype;
    r_ubic c_th_ubic%rowtype;
    r_etnia c_th_etnia%rowtype; 

    --------------------
    v_id_dim_tiempo int := 0 ; 
    v_num_iter int := 0 ;
    v_carga int := 0 ;
    r1 int ;
    r2 int ;
    BEGIN 
-- Verifica que no se tenga cargado el periodo solicitado a traves de los parámetros de entrada año y trimerstre de la función
select ID_DIMT into v_id_dim_tiempo
 from d_tiempo 
 where anio = vanio and trimestre = vtrimestre and mes is null;
select count(*) into v_carga
 from th_defuncionesfetales 
 where ID_DIMTIEMPO = v_id_dim_tiempo;
----- determinar rango de mes de inicio y fin del trimestre ingresado
if vtrimestre =1 then 
        r1:=1;
        r2:=3;
    elsif vtrimestre=2 then
        r1:=4;
        r2:=6;
    elsif vtrimestre=3 then
        r1:=7;
        r2:=9;
    elsif vtrimestre=4 then
        r1:=10;
        r2:=12;
end if ;
if v_carga > 0 then
   RETURN v_num_iter;
end if;
-------- Análisis en base a la dimensión tiempo
OPEN c_th_tiempo;
fetch c_th_tiempo into r_tiempo; -- campos de la dimension tiempo

while c_th_tiempo%found loop

    if vtrimestre = 4 then
       insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                   select r_tiempo.id_dimt,null,null,estado_civil,sexo,count(*)
                     from tg_defuncionesfetales 
                    where to_number(anio_fall) = r_tiempo.anio
                 group by estado_civil,sexo
                 having count(*) > 0;
                 commit;
    elsif r_tiempo.trimestre is not null and r_tiempo.mes is null then
       insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                   select r_tiempo.id_dimt,null,null,estado_civil,sexo,count(*)
                     from tg_defuncionesfetales 
                    where to_number(anio_fall) = r_tiempo.anio
                    and mes_fall between r1 and r2
                 group by estado_civil,sexo
                 having count(*) > 0;
                 commit;

    elsif r_tiempo.trimestre is not null and r_tiempo.mes is not null then  
          insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                   select r_tiempo.id_dimt,null,null,estado_civil,sexo,count(*)
                     from tg_defuncionesfetales 
                    where to_number(anio_fall) = r_tiempo.anio
                    and mes_fall = r_tiempo.mes
                 group by estado_civil,sexo
                 having count(*) > 0;  
                 commit;
    end if;

    -------- Análisis en base a la dimensión tiempo + ubicación geográfica
    OPEN c_th_ubic;

    fetch c_th_ubic into r_ubic; -- campos de la dimension tiempo

    while c_th_ubic%found loop

        if vtrimestre = 4 then
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                       group by estado_civil,sexo
                       having count(*) > 0;
                       commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                       group by estado_civil,sexo
                       having count(*) > 0;
                       commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                       group by estado_civil,sexo
                       having count(*) > 0;
                       commit;

           end if;            

       elsif r_tiempo.trimestre is not null and r_tiempo.mes is null then
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                       select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                         from tg_defuncionesfetales 
                        where to_number(anio_fall) = r_tiempo.anio
                          and mes_fall between r1 and r2
                          and prov_fall = r_ubic.id_provincia
                         group by estado_civil,sexo
                         having count(*) > 0;
              commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall between r1 and r2
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                       group by estado_civil,sexo
                       having count(*) > 0;
               commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall between r1 and r2
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                       group by estado_civil,sexo
                       having count(*) > 0;
               commit;

           end if;            

        elsif r_tiempo.trimestre is not null and r_tiempo.mes is not null then  
           -- analiza por año - mes - provincia     
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
                  insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                           select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                             from tg_defuncionesfetales 
                            where to_number(anio_fall) = r_tiempo.anio
                              and mes_fall = r_tiempo.mes
                              and prov_fall = r_ubic.id_provincia
                             group by estado_civil,sexo
                             having count(*) > 0;
                  commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              -- analiza por año - mes - provincia - canton
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall = r_tiempo.mes
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                       group by estado_civil,sexo
                       having count(*) > 0;
              commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              -- analiza por año - mes - provincia - canton - parroquia
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,null,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall = r_tiempo.mes
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                       group by estado_civil,sexo
                       having count(*) > 0;
                commit;

           end if;  

        end if;

        -------- Análisis en base a la dimensión tiempo + ubicación geográfica + dimension etnia --------------------

        OPEN c_th_etnia;
        fetch c_th_etnia into r_etnia; -- campos de la dimension etnia
        while c_th_etnia%found loop

    if vtrimestre = 4 then
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
               commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
               commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
               commit;

           end if;            

       elsif r_tiempo.trimestre is not null and r_tiempo.mes is null then
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                       select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                         from tg_defuncionesfetales 
                        where to_number(anio_fall) = r_tiempo.anio
                          and mes_fall between r1 and r2
                          and prov_fall = r_ubic.id_provincia
                          and etnia = r_etnia.id_dimetnia
                         group by estado_civil,sexo
                         having count(*) > 0;
              commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall between r1 and r2
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
              commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall between r1 and r2
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
                commit;
           end if;            

        elsif r_tiempo.trimestre is not null and r_tiempo.mes is not null then  
           -- analiza por año - mes - provincia -etnia    
           if r_ubic.id_parroquia is null and r_ubic.id_canton is null then
                  insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA,EST_CIVIL,SEXO, TOTAL) 
                           select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                             from tg_defuncionesfetales 
                            where to_number(anio_fall) = r_tiempo.anio
                              and mes_fall = r_tiempo.mes
                              and prov_fall = r_ubic.id_provincia
                              and etnia = r_etnia.id_dimetnia
                             group by estado_civil,sexo
                             having count(*) > 0;
                   commit;

           elsif r_ubic.id_parroquia is null and r_ubic.id_canton is not null then
              -- analiza por año - mes - provincia - canton -etnia
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall = r_tiempo.mes
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
                 commit;

           elsif r_ubic.id_parroquia is not null and r_ubic.id_canton is not null then
              -- analiza por año - mes - provincia - canton - parroquia -etnia
              insert into th_defuncionesfetales (ID_DIMTIEMPO,ID_DIMUBIC,ID_DIMETNIA, EST_CIVIL,SEXO, TOTAL) 
                         select r_tiempo.id_dimt,r_ubic.id_dimubic,r_etnia.id_dimetnia,estado_civil,sexo,count(*)
                           from tg_defuncionesfetales 
                          where to_number(anio_fall) = r_tiempo.anio
                            and mes_fall = r_tiempo.mes
                            and prov_fall = r_ubic.id_provincia
                            and cant_fall = r_ubic.id_canton
                            and parr_fall = r_ubic.id_parroquia
                            and etnia = r_etnia.id_dimetnia
                       group by estado_civil,sexo
                       having count(*) > 0;
            commit;
           end if;  

        end if;
        fetch c_th_etnia into r_etnia;
        end loop; --Etnia 
        close c_th_etnia; --- se cierra el cursor de dimension etnia

-------------------------------------------------------------------
        fetch c_th_ubic into r_ubic; 
        v_num_iter := v_num_iter +1;
    end loop; -- Ubicacion geográfica
    close  c_th_ubic;
    fetch c_th_tiempo into r_tiempo;
end loop;  -- Tiempo
close  c_th_tiempo;
commit;
RETURN v_num_iter;
END;
