  AS select tiem.anio||' '|| tiem.trimestre "Año Trim", ub.nombre_provincia ||' '|| ub.nombre_canton 
  "Provincia Canton",dl.descripcion_lugar "Lugar Ocurrencia",sum (th.total) "Número de muertes"
